**基本操作**  
```
//查看所有数据库
show databases;  

//创造数据库
create database 数据库名;   

//使用自己的数据库
use 数据库名;      

//查看表格
show tables;   //要求前面有use语句
show tables from 数据库名;

//创造表格
create table 表名称(
字段名  数据类型,
字段名 数据类型
);

//查看表的数据
select * from 数据库表名称;

//添加记录
insert into 表名称 values(值列表);
insert into student values(1,'张三');

//查看表的创建信息
show create table 表名称\G

//查看数据库的创建信息
show create database 数据库名\G

//删除表格
drop table 表名称;

//删除数据库
drop database 数据库名;
```
